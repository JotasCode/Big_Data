# Importamos la libreria datetime
from datetime import date 

# Declaramos la clase para calcular la edad de una persona.
# Una clase es un plano en el que definimos los atributos y metodos con el que seran creados los objetos.
# Un atributo es una caracteristica (valores en variables) de un objeto.
# Un metodo es el conportamiento de un objeto (las acciones o funciones que se pueden hacer).

class CalculadoraEdad:
    # La función "__init__" es la función constructora que debe estar de primero.
    # La variable global "self" dabe ir como parametro en cada metodo a definir en la clase def __init__(self, anio_nacimiento, mes_nacimiento, dia_nacimiento):

        self.fecha_nacimiento = date(anio_nacimiento, mes_nacimiento, dia_nacimiento)

        return self.fecha_nacimiento

    def calcular_edad(self):
    hoy = date.today()
    edad = hoy.year - self.fecha_nacimiento.year

    if (hoy.month, hoy.day) < (self.fecha_nacimiento.month, self.fecha_nacimiento.day):
        edad -= 1

    return edad

#----------------- Implementación --------------------
anio = 1990
mes = 5
dia = 15

persona1 = CalculadoraEdad(anio, mes, dia)
edd_actual = persona1.calcular_edad()

print(f"Fecha de nacimiento: {anio}-{mes:02d}-{dia:02d}")
print(f"La edad actual es: {edad_actual} años.")
